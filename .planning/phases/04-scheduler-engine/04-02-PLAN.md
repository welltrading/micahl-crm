    ```typescript
    import { sendWhatsAppMessage } from '../../lib/airtable/green-api';
    import { createMessageLogEntry } from '../../lib/airtable/message-log';
    import { normalizePhone } from '../../lib/airtable/phone';
    ```

    The run() function logic:
    1. `const now = new Date().toISOString()`
    2. `const due = await getPendingMessagesDue(now)` — if empty, return
    3. For each scheduled message:
       a. `await markMessageSending(msg.id)` — CLAIM before sending (idempotency lock)
       b. `const enrollments = await getEnrollmentsForCampaign(msg.campaign_id[0])`
       c. For each enrollment:
          - `const contact = await getContactById(enrollment.contact_id[0])`
          - if no contact: log failed entry, continue
          - `const chatId = normalizePhone(contact.phone) + '@c.us'`
          - try: `await sendWhatsAppMessage(chatId, msg.message_content)`
          - on success: `await createMessageLogEntry({ ..., status: 'sent', green_api_response: result.idMessage })`
          - on error: `await createMessageLogEntry({ ..., status: 'failed', error_message: String(err) })`
          - 500ms delay between sends: `await new Promise(r => setTimeout(r, 500))`
       d. After all contacts: `await markMessageSent(msg.id)` — only after all contacts processed
    4. `run().catch(console.error)` at module bottom
