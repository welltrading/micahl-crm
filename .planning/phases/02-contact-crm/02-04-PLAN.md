    ```typescript
    export async function getContactDetail(contactId: string) {
      const [enrollments, messages] = await Promise.all([
        getContactEnrollments(contactId),
        getContactMessages(contactId),
      ]);
      return { enrollments, messages };
    }
    ```

    **Create src/components/contacts/ContactDetailPanel.tsx** ('use client'):

    Props: `{ contact: Contact | null; onClose: () => void }`

    State:
    - `detail: { enrollments: CampaignEnrollment[]; messages: ScheduledMessage[] } | null`
    - `loading: boolean`

    Behavior:
    - When `contact` changes to a non-null value (useEffect on `contact?.id`): set loading=true, call getContactDetail(contact.id), set detail, set loading=false
    - When contact is null: reset detail to null

    Render structure:
    ```
    <Sheet open={!!contact} onOpenChange={(o) => !o && onClose()}>
      <SheetContent side="left">
        <SheetHeader>
          <SheetTitle>{contact?.full_name}</SheetTitle>
        </SheetHeader>

        // Contact info section
        <p>טלפון: {formatPhoneDisplay(contact.phone)}</p>
        <p>תאריך הצטרפות: {contact.joined_date formatted as he-IL locale date}</p>

        // Message history section
        if (loading) → show "טוען..." text
        if (!detail || detail.enrollments.length === 0) → show "אין רישומים לקמפיינים"
        else → render campaign-grouped message list:
          For each enrollment in detail.enrollments:
            - Show campaign section header (use enrollment.campaign_id[0] as key; show "קמפיין: {id}" for now — campaign name will be enhanced in Phase 3)
            - Filter detail.messages where message.campaign_id[0] === enrollment.campaign_id[0]
            - For each message: show offset_label (Hebrew), status (Hebrew badge), send_at (formatted date)
            - Empty message list for a campaign: show "אין הודעות מתוזמנות"
    ```
