```typescript
function formatSendPreview(send_at_utc: string | null): string {
  if (!send_at_utc) return '';
  const d = new Date(send_at_utc);
  const date = d.toLocaleDateString('he-IL', { timeZone: 'Asia/Jerusalem', day: 'numeric', month: 'numeric' });
  const time = d.toLocaleTimeString('he-IL', { timeZone: 'Asia/Jerusalem', hour: '2-digit', minute: '2-digit', hour12: false });
  return `יישלח ב-${date} בשעה ${time}`;
}
```

```typescript
const TIME_OPTIONS = Array.from({ length: 48 }, (_, i) => {
  const h = Math.floor(i / 2).toString().padStart(2, '0');
  const m = i % 2 === 0 ? '00' : '30';
  return `${h}:${m}`;
});
```
